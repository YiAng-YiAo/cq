--[[脚本主函数]]--
function main(sysarg)
  local flag = false
  for _,conf in pairs(EscortConf.Quests) do
    if LActor.hasQuest(sysarg, conf.questId) then
      flag = true
      break
    end
  end
  if flag then
    local pack = LDataPack.allocPacket(sysarg, SystemId.miscsSystem, MiscsSystemProtocol.sEndEscort)
    if pack then
      LDataPack.flush(pack)
    end
  else
  	return commonQuestNpcMain(sysarg)
  end
end

--[[初始化函数]]--
function initialization(npcobj)
  local i
  thisNPC = npcobj
  --SPID = TSystem.getGlobalVariant( "spid" )
  for i = 1, table.getn(InitFnTable) do
    InitFnTable[i]( npcobj )
  end
end

--[[析构化函数]]--
function finalization(npcobj)
  local i
  for i = 1, table.getn(FinaFnTable) do
    FinaFnTable[i]( npcobj )
  end
  thisNPC = nil
end
